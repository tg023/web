Form Validation

<!DOCTYPE html>
<html lang="en" ng-app="formApp">
<head>
  <meta charset="UTF-8">
  <title>Simple Angular Form Validation</title>
  <script src="https://cdnjs.cloudflare.com/ajax/libs/angular.js/1.8.3/angular.min.js"></script>
</head>
<body ng-controller="FormController">

  <h2>Simple Registration Form</h2>

  <form name="userForm" novalidate ng-submit="submitForm()">

    <!-- Name -->
    <label>Name:</label><br>
    <input type="text" name="name" ng-model="user.name" ng-required="true">
    <span ng-show="userForm.name.$dirty && userForm.name.$error.required" style="color:red">
      Name is required.
    </span>
    <br><br>

    <!-- Email -->
    <label>Email:</label><br>
    <input type="email" name="email" ng-model="user.email" ng-required="true">
    <span ng-show="userForm.email.$dirty && userForm.email.$error.required" style="color:red">
      Email is required.
    </span>
    <span ng-show="userForm.email.$dirty && userForm.email.$error.email" style="color:red">
      Invalid email format.
    </span>
    <br><br>

    <!-- Password -->
    <label>Password:</label><br>
    <input type="password" name="password" ng-model="user.password" ng-required="true" ng-minlength="6">
    <span ng-show="userForm.password.$dirty && userForm.password.$error.required" style="color:red">
      Password is required.
    </span>
    <span ng-show="userForm.password.$dirty && userForm.password.$error.minlength" style="color:red">
      Must be at least 6 characters.
    </span>
    <br><br>

    <!-- Confirm Password -->
    <label>Confirm Password:</label><br>
    <input type="password" name="confirmPassword" ng-model="user.confirmPassword" ng-required="true">
    <span ng-show="userForm.confirmPassword.$dirty && userForm.confirmPassword.$error.required" style="color:red">
      Please confirm your password.
    </span>
    <span ng-show="userForm.confirmPassword.$dirty && user.password !== user.confirmPassword" style="color:red">
      Passwords do not match.
    </span>
    <br><br>

    <!-- Age -->
    <label>Age:</label><br>
    <input type="number" name="age" ng-model="user.age" ng-required="true" min="13" max="120">
    <span ng-show="userForm.age.$dirty && userForm.age.$error.required" style="color:red">
      Age is required.
    </span>
    <span ng-show="userForm.age.$dirty && userForm.age.$error.min" style="color:red">
      Must be at least 13.
    </span>
    <span ng-show="userForm.age.$dirty && userForm.age.$error.max" style="color:red">
      Must be below 120.
    </span>
    <br><br>

    <!-- Country -->
    <label>Country:</label><br>
    <select name="country" ng-model="user.country" ng-required="true">
      <option value="">Select country</option>
      <option value="India">India</option>
      <option value="USA">USA</option>
      <option value="UK">UK</option>
    </select>
    <span ng-show="userForm.country.$dirty && userForm.country.$error.required" style="color:red">
      Please select your country.
    </span>
    <br><br>

    <!-- Terms -->
    <label>
      <input type="checkbox" name="terms" ng-model="user.terms" ng-required="true">
      I agree to terms
    </label>
    <span ng-show="userForm.terms.$dirty && userForm.terms.$error.required" style="color:red">
      You must accept the terms.
    </span>
    <br><br>

    <!-- Submit Button -->
    <button type="submit" ng-disabled="userForm.$invalid || user.password !== user.confirmPassword">
      Submit
    </button>
  </form>

  <hr>

  <!-- Show submitted data -->
  <div ng-if="submitted">
    <h3>Form Submitted Successfully!</h3>
    <p><b>Name:</b> {{user.name}}</p>
    <p><b>Email:</b> {{user.email}}</p>
    <p><b>Age:</b> {{user.age}}</p>
    <p><b>Country:</b> {{user.country}}</p>
  </div>

  <script>
    angular.module('formApp', [])
    .controller('FormController', function($scope) {
      $scope.user = {};

      $scope.submitForm = function() {
        if ($scope.userForm.$valid && $scope.user.password === $scope.user.confirmPassword) {
          $scope.submitted = true;
          console.log('Submitted:', $scope.user);
        } else {
          alert('Please fix all errors before submitting.');
        }
      };
    });
  </script>

</body>
</html>
